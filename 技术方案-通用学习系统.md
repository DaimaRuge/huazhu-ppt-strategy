### 4.2 学科规则引擎

#### 4.2.1 规则基类
```python
from abc import ABC, abstractmethod

class SubjectRule(ABC):
    def __init__(self, stage, config):
        self.stage = stage
        self.config = config
    
    @abstractmethod
    def generate_question(self, topic, difficulty):
        pass
    
    @abstractmethod
    def evaluate_answer(self, question, answer):
        pass
    
    @abstractmethod
    def get_review_logic(self):
        pass
```

#### 4.2.2 语文学科规则
```python
class ChineseRule(SubjectRule):
    def generate_question(self, topic, difficulty):
        if self.stage == "kindergarten":
            return self._generate_kindergarten(topic, difficulty)
        elif self.stage == "elementary":
            return self._generate_elementary(topic, difficulty)
        elif self.stage == "junior_high":
            return self._generate_junior_high(topic, difficulty)
    
    def _generate_kindergarten(self, topic, difficulty):
        if topic == "literacy_game":
            return {
                "type": "image_match",
                "character": "山",
                "images": ["山", "水", "火", "土"],
                "hint": "找一找哪个是'山'字？"
            }
    
    def _generate_elementary(self, topic, difficulty):
        if topic == "characters":
            return {
                "type": "dictation",
                "words": ["美丽", "祖国", "骄傲"],
                "hint": "请听写下面的词语"
            }
        elif topic == "writing":
            return {
                "type": "picture_writing",
                "image": "park_scene.jpg",
                "requirements": [
                    "写一段话",
                    "不少于100字",
                    "用到3个形容词"
                ]
            }
    
    def _generate_junior_high(self, topic, difficulty):
        if topic == "modern_reading":
            return {
                "type": "reading_comprehension",
                "text": "...",
                "questions": [
                    "请概括本文的主要内容",
                    "分析第三段的作用",
                    "谈谈你对最后一句话的理解"
                ]
            }
    
    def evaluate_answer(self, question, answer):
        # 分阶段评分逻辑
        if self.stage == "kindergarten":
            return self._evaluate_kindergarten(question, answer)
        elif self.stage == "elementary":
            return self._evaluate_elementary(question, answer)
        elif self.stage == "junior_high":
            return self._evaluate_junior_high(question, answer)
```

#### 4.2.4 英语学科规则
```python
class EnglishRule(SubjectRule):
    def _generate_kindergarten(self, topic, difficulty):
        if topic == "alphabet":
            return {
                "type": "letter_match",
                "letter": "A",
                "options": ["A", "B", "C", "D"],
                "audio": "play letter A sound"
            }
    
    def _generate_elementary(self, topic, difficulty):
        if topic == "vocabulary":
            return {
                "type": "sentence_making",
                "word": "beautiful",
                "requirements": [
                    "造一个完整的句子",
                    "8-12个词",
                    "时态正确"
                ]
            }
    
    def _generate_junior_high(self, topic, difficulty):
        if topic == "reading":
            return {
                "type": "cloze_test",
                "text": "...",
                "blanks": [1, 5, 8, 12, 15],
                "options": {...}
            }
    
    def lemmatize_word(self, word):
        # 单词形态归并
        irregular = {
            "received": "receive",
            "chuckled": "chuckle",
            "staring": "stare"
        }
        if word in irregular:
            return irregular[word]
        # 后缀去除
        suffixes = ["ed", "ing", "s", "es"]
        for suffix in suffixes:
            if word.endswith(suffix):
                return word[:-len(suffix)]
        return word
```

### 4.3 多阶段激励系统

#### 4.3.1 星星奖励系统（幼儿园）
```python
class StarRewardSystem:
    def __init__(self, child_id):
        self.child_id = child_id
        self.stars = 0
        self.stickers = []
    
    def complete_task(self, task_type, performance):
        stars_earned = 1
        if performance == "excellent":
            stars_earned += 1  # 额外奖励
        
        self.stars += stars_earned
        
        # 随机贴纸
        if random.random() < 0.3:
            sticker = self._get_random_sticker()
            self.stickers.append(sticker)
        
        return {
            "stars_earned": stars_earned,
            "total_stars": self.stars,
            "sticker": sticker if 'sticker' in locals() else None,
            "message": f"太棒了！获得 {stars_earned} 颗星星 ⭐"
        }
    
    def _get_random_sticker(self):
        stickers = ["🌟", "🎉", "🏆", "🌈", "🦋", "🚀", "🌸", "🐱"]
        return random.choice(stickers)
```

#### 4.3.2 积分系统（小学）
```python
class PointsSystem:
    def __init__(self, child_id):
        self.child_id = child_id
        self.total_points = 0
        self.consecutive_days = 0
    
    def complete_exercise(self, is_correct):
        points = 5  # 基础分
        
        if is_correct:
            points += 3  # 全对额外分
        
        # 连续学习奖励
        points += min(self.consecutive_days, 7)
        
        self.total_points += points
        return {
            "earned": points,
            "total": self.total_points,
            "consecutive_days": self.consecutive_days,
            "message": f"本题 +{points} 分，累计 {self.total_points} 分"
        }
```

class AchievementSystem:
    def __init__(self, child_id):
        self.child_id = child_id
        self.achievements = []
        self.milestones = []
        self.badges = []
    
    def complete_mastery(self, topic, mastery_level):

            achievement = {
                "name": f"{topic} 精通",
                "type": "mastery",
                "timestamp": datetime.now()
            }
            self.achievements.append(achievement)
            return achievement
        return None
    
    def check_milestone(self, streak_days):
        milestones = [7, 30, 100, 365]
        if streak_days in milestones and streak_days not in self.milestones:
            self.milestones.append(streak_days)
            return {
                "type": "streak",
                "days": streak_days,
                "badge": f"连续{streak_days}天"
            }
        return None
